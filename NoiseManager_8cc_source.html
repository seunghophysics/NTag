143void NoiseManager::SetNoiseTreeFromOptions(TString option, int nInputEvents, float tStart, float tEnd, int seed)
144{
145 fNoiseType = option;
146 SetNoiseTimeRange(tStart, tEnd);
147
148 // Read dummy (TChain)
149 TChain* dummyChain = new TChain(fNoiseTreeName);
150
151 TObjArray* opt = option.Tokenize(':');
152 TString base = ((TObjString*)(opt->At(0)))->GetString();
153 TString run = "";
154 int minRun = -1;
155 int maxRun = 1000000;
156 if (option.Contains(":")) {
157 run = "0" + ((TObjString*)(opt->At(1)))->GetString();
158 if (run.Contains("-")) {
159 TObjArray* runRange = run.Tokenize('-');
160 minRun = ((TObjString*)(runRange->At(0)))->GetString().Atoi();
161 maxRun = ((TObjString*)(runRange->At(1)))->GetString().Atoi();
162 run = "";
163 }
164 }
165
166 SetSeed(seed);
167 TString dummyRunPath = fNoisePath + "/" + option;
168 int nRequiredEvents = nInputEvents / fNParts;
169
170 if (base == "default") base = "sk" + std::to_string(fSKGen);
171 std::vector<TString> runDirs = GetListOfSubdirectories(fNoisePath + "/" + base);
172 std::vector<TString> fileList;
173 if (run != "") fileList = GetListOfFiles(fNoisePath + "/" + base + "/" + run);
174
175 TString dummyFilePath;
176
177 while (fNEntries <= /*2**/nRequiredEvents) {
178 if (run == "") {
179 dummyRunPath = PickRandom(runDirs);
180 int dummyRunNo = ((TObjString*)((dummyRunPath.Tokenize('/'))->Last()))->GetString().Atoi();
181 if (!dummyRunNo) continue; // skip any non-numeric run directory name
182 if (dummyRunNo < minRun || maxRun < dummyRunNo ) continue;
183 fileList = GetListOfFiles(dummyRunPath, ".root");
184 }
185 if (!fileList.empty())
186 dummyFilePath = PickRandom(fileList);
187 else continue;
188 AddNoiseFileToChain(dummyChain, dummyFilePath);
189 }
190
191 SetNoiseTree(dummyChain);
192}

194void NoiseManager::SetNoiseTreeFromList(TString pathToList)
195{
196 std::ifstream inFile;
197 inFile.open(pathToList.Data());
198
199 // read in settings
200 std::string option; float value;
201 float startTime = fNoiseStartTime*1e-3 - 1;
202 float endTime = fNoiseEndTime*1e-3 - 1;
203 while (inFile >> option >> value) {
204 if (option=="TNOISESTART") startTime = value;
205 if (option=="TNOISEEND") endTime = value;
206 if (option=="PMTDEADTIME") fPMTDeadtime = value;
207 if (option=="noise_cut") fDoN200Cut = value;

209 if (option=="ODMAXN200") fODMaxN200 = value;
210 }
211 SetNoiseTimeRange(startTime, endTime);
212
213 inFile.clear();
214 inFile.seekg(0);
215
216 // read in files
217 TChain* dummyChain = new TChain(fNoiseTreeName);
218 std::string dummyFilePath;
219 while (std::getline(inFile, dummyFilePath)) {
220 if (dummyFilePath[0]=='/')
221 AddNoiseFileToChain(dummyChain, dummyFilePath);
222 }
223
224 inFile.close();
225 SetNoiseTree(dummyChain);

283void NoiseManager::GetNextNoiseEvent()
284{
285 fPartID = 0; fCurrentIDHitIndex = 0; fCurrentODHitIndex = 0;
286
287 fCurrentEntry++; fIDNoiseEventHits.Clear(); fODNoiseEventHits.Clear();
288 if (fCurrentEntry < fNEntries) {
289 fNoiseTree->GetEntry(fCurrentEntry);
290 int trgType = fHeader->idtgsk;
291 if (trgType & mRandomWide || trgType == mT2KDummy || trgType & mNickel)
292 SetNoiseEventHits();
293 else
294 GetNextNoiseEvent();
295 }
296 else {
297 fMsg.Print("Noise tree reached its end!", pWARNING);
298 if (fCurrentEntry == fNEntries && fDoRepeat) {
299 // start from beginning
300 fMsg.Print("Repetition allowed: going back to the first entry in noise tree...", pWARNING);
301 fCurrentEntry = 0;
302 fNoiseTree->GetEntry(fCurrentEntry);
303 }
304 else {
305 // error
306 fMsg.Print("Repetition disallowed. To allow, use NoiseManager::SetRepeat(true). Aborting program...", pERROR);
307 }
308 }

311void NoiseManager::SetNoiseEventHits()
312{
313 bool useThisNoiseEvent = true;
314
315 if (fDoN200Cut) {
316 // dark selection: OD max N200 <= 20 && ID max N200 <= 50
317 auto idHist = Histogram(fIDTQReal->T, 5000, -500e3, 500e3);
318 auto odHist = Histogram(fODTQReal->T, 5000, -500e3, 500e3);
319 for (int iBin=0; iBin<5000; iBin++) {
320 if (odHist[iBin].second > fODMaxN200 | idHist[iBin].second > fIDMaxN200) {
321 fMsg.Print(Form("Rejecting noise event with OD N200 %d and ID N200 %d...",
322 odHist[iBin].second, idHist[iBin].second), pWARNING);
323 useThisNoiseEvent = false; break;
324 }
325 }
326 }
327
328 // make sure noise event is not empty
329 if (!fIDTQReal->nhits || !fODTQReal->nhits) {
330 fMsg.Print(Form("Skipping an empty noise event..."), pWARNING);
331 useThisNoiseEvent = false;
332 }
333
334 // populate hit clusters
335 if (useThisNoiseEvent) {
336 PopulateHitCluster(&fIDNoiseEventHits);
337 PopulateHitCluster(&fODNoiseEventHits, true);
338
339 int nHits = fIDNoiseEventHits.GetSize();
340 fNoiseEventLength = fIDNoiseEventHits[nHits-1].t() - fIDNoiseEventHits[0].t();
341 fNParts = (int)(fNoiseEventLength / fNoiseWindowWidth);
342 fNoiseT0 = fIDNoiseEventHits[0].t() + (fNoiseEventLength - fNParts*fNoiseWindowWidth)*ranGen.Uniform();
343
344 if (fNoiseEventLength < fNoiseWindowWidth) {
345 fMsg.Print(Form("Noise event length %3.2f us is smaller than required window width %3.2f us, "
346 "getting next noise event...",
347 fNoiseEventLength*1e-3, fNoiseWindowWidth*1e-3), pWARNING);
348 GetNextNoiseEvent();
349 }
350 }
351
352 else GetNextNoiseEvent();
353}

355void NoiseManager::AddNoise(PMTHitCluster* signalHits, PMTHitCluster* noiseHits, int& currentHitIndex, float darkRate, bool OD)
356{
357 // noise from noise files
358 if (fNoiseTree) {
359
360 if (fCurrentEntry == -1 || fPartID == fNParts) {
361 GetNextNoiseEvent();
362 fMsg.Print(Form("Current noise entry: %d", fCurrentEntry), pDEBUG);
363 }
364 //std::cout << "NoiseHitsSize: " << noiseHits->GetSize() << "\n";
365 //std::cout << "CurrentHitTime: " << noiseHits->At(currentHitIndex).t() << " Index: " << currentHitIndex << "\n";
366
367 float partStartTime = fNoiseT0 + fPartID * fNoiseWindowWidth;
368 float partEndTime = partStartTime + fNoiseWindowWidth;
369
370 while (noiseHits->At(currentHitIndex).t() < partStartTime) currentHitIndex++;
371
372 while (noiseHits->At(currentHitIndex).t() < partEndTime) {
373 PMTHit hit = noiseHits->At(currentHitIndex);
374 hit += (fNoiseStartTime - partStartTime);
375 signalHits->Append(hit);
376 currentHitIndex++;
377 }
378 }
379
380 // in case fNoiseTree is empty, simulate noise
381 else {
382 int nSegments = int(fNoiseWindowWidth / fPMTDeadtime);
383 float expected = darkRate * 1e3 * fPMTDeadtime * 1e-9;
384
385 unsigned int iMinPMT = !OD? 1 : 20001;
386 unsigned int iMaxPMT = !OD? MAXPM : 20000+MAXPMA;
387
388 for (unsigned int iPMT=iMinPMT; iPMT<=iMaxPMT; iPMT++) {
389 for (unsigned int iSeg=0; iSeg<=nSegments; iSeg++) {
390 int sampled = ranGen.Poisson(expected);
391 if (sampled) {
392 float hitT = fNoiseStartTime + (iSeg + ranGen.Uniform()) * fPMTDeadtime;
393 if (hitT < fNoiseEndTime) {
394 float hitQ = abs(ranGen.Gaus(1, 0.7));
395 PMTHit hit(hitT, hitQ, iPMT, 2/* in-gate */);
396 signalHits->Append(hit);
397 }
398 }
399 }
400 }
401 }
402 //fMsg.Print("ApplyDeadtime: ", pDEFAULT, false);
403 //signalHits->ApplyDeadtime(fPMTDeadtime, true);
404 signalHits->Sort();
405 //signalHits->CheckNaN();
406 fPartID++;
407}

409void NoiseManager::PopulateHitCluster(PMTHitCluster* hitCluster, bool OD)
410{
411 std::vector<float> t = !OD? fIDTQReal->T : fODTQReal->T;
412 std::vector<float> q = !OD? fIDTQReal->Q : fODTQReal->Q;
413 std::vector<int> i = !OD? fIDTQReal->cables : fODTQReal->cables;
414
415 unsigned int nRawHits = t.size();
416
417 assert((t.size()==q.size()) && (q.size()==i.size()));
418
419 for (unsigned int j=0; j<nRawHits; j++) {
420 if (-1000e3 < t[j] && t[j] < 1000e3) {
421 hitCluster->Append({t[j], q[j], i[j]&0x0000FFFF, 2/*in-gate flag*/});
422 }
423 else
424 fMsg.Print(Form("Skipping hit with time T=%3.2f msec which is outside of range [-1, 1] msec...", t[j]*1e-6), pDEBUG);
425 }
426
427 hitCluster->Sort();
428}
