173void NoiseManager::SetNoiseEventHits()
174{
175 fT = fTQReal->T;
176 fQ = fTQReal->Q;
177 fI = fTQReal->cables;
178 unsigned int nRawHits = fT.size();
179
180 for (unsigned int j=0; j<=nRawHits; j++) {
181 if (-1000e3 < fT[j] && fT[j] < 1000e3)
182 fNoiseEventHits.Append({fT[j], fQ[j], fI[j]&0x0000FFFF, 2/*in-gate flag*/});
183 else
184 fMsg.Print(Form("Skipping hit with time T=%3.2f msec which is outside of range [-1, 1] msec...", fT[j]*1e-6), pWARNING);
185 }
186
187 fNoiseEventHits.Sort();
188 int nHits = fNoiseEventHits.GetSize();
189 fNoiseEventLength = fNoiseEventHits[nHits-1].t() - fNoiseEventHits[0].t();
190 float rawHitDensity = nRawHits / 1000.;
191 fNParts = (int)(fNoiseEventLength / fNoiseWindowWidth);
192
193 if (fNParts <= 0 || rawHitDensity < fMinHitDensity || rawHitDensity > fMaxHitDensity) {
194 fMsg.Print(Form("Skipping inappropriate entry with fNParts = %d, rawHitDensity = %3.2f hits/us\n", fNParts, rawHitDensity), pWARNING, false);
195 GetNextNoiseEvent();
196 }
197
198 fNoiseT0 = fT[0] + (fNoiseEventLength - fNParts*fNoiseWindowWidth)/2.;
199}

201void NoiseManager::AddNoise(PMTHitCluster* signalHits)
202{
203 if (fCurrentEntry == -1 || fPartID == fNParts) {
204 GetNextNoiseEvent();
205 fMsg.Print(Form("Current noise entry: %d", fCurrentEntry), pDEBUG);
206 }
207
208 float partStartTime = fNoiseT0 + fPartID * fNoiseWindowWidth;
209 float partEndTime = partStartTime + fNoiseWindowWidth;
210
211 while (fNoiseEventHits[fCurrentHitID].t() < partStartTime) fCurrentHitID++;
212
213 while (fNoiseEventHits[fCurrentHitID].t() < partEndTime) {
214 PMTHit hit = fNoiseEventHits[fCurrentHitID];
215 hit += (fNoiseStartTime - partStartTime);
216 signalHits->Append(hit);
217 fCurrentHitID++;
218 }
219
220 signalHits->ApplyDeadtime(fPMTDeadtime);
221 signalHits->Sort();
222 fPartID++;
223}
