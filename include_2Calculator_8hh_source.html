118template <typename T>
119float GetRMS(const std::vector<T>& vec)
120{
121 float N = static_cast<float>(vec.size());
122 float mean = 0.;
123 float var = 0.;
124
125 for (auto const& value: vec)
126 mean += value / N;
127 for (auto const& value: vec)
128 var += (value-mean)*(value-mean) / (N-1);
129
130 return sqrt(var);
131}

156template <typename T>
157float GetSkew(const std::vector<T>& vec)
158{
159 float m3 = 0;
160 float mean = GetMean(vec);
161 float N = vec.size();
162
163 for (auto const& value: vec) {
164 m3 += pow((value - mean), 3.);
165 }
166 m3 /= N;
167
168 return m3 / pow(GetRMS(vec), 1.5);
169}
