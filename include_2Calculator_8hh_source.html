122template <typename T>
123float GetRMS(const std::vector<T>& vec)
124{
125 float N = static_cast<float>(vec.size());
126 float mean = 0.;
127 float var = 0.;
128
129 for (auto const& value: vec)
130 mean += value / N;
131 for (auto const& value: vec)
132 var += (value-mean)*(value-mean) / (N-1);
133
134 return sqrt(var);
135}
