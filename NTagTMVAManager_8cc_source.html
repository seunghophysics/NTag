24void NTagTMVAManager::InitializeReader(std::string weightPath)
25{
26 // use same weight for bonsai and lowfit
27 if (weightPath == "lowfit") weightPath = "bonsai";
28 if (weightPath == "bonsai" || weightPath == "trms" || weightPath == "prompt")
29 SetWeightPath(GetENV("NTAGLIBPATH")+ "weights/tmva/" + weightPath + "/NTagTMVAFactory_MLP.weights.xml");
30 else if (!weightPath.empty())
31 SetWeightPath(weightPath);
32
33 fReader = new TMVA::Reader();
34
35 for (auto const& feature: gTMVAFeatures) {
36 fFeatureContainer[feature] = 0;
37 fReader->AddVariable(feature, &(fFeatureContainer[feature]));
38 }
39
40 fReader->AddSpectator("Label", &(fCandidateLabel));
41 fReader->BookMVA("MLP", fWeightFilePath);
42}
