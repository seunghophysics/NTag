161unsigned int PMTHitCluster::RemoveBadChannels()

175
176 return nSize-GetSize();

329PMTHitCluster PMTHitCluster::SliceRange(Float startT, Float lowT, Float upT)
330{

332 if (fElement.empty())
333 return selectedHits;

338 std::cerr << "PMTHitCluster::Slice : lower bound is larger than upper bound." << std::endl;
339
340 PMTHit startHit(startT, 0, 1, 1);
341 unsigned int low = GetLowerBoundIndex(startT + lowT);
342 unsigned int up = GetUpperBoundIndex(startT + upT);
343
344 if (fHasVertex)
345 selectedHits.SetVertex(fVertex);
346
347 for (unsigned int iHit = low; iHit <= up; iHit++)
348 selectedHits.Append(fElement.at(iHit));
349
350 return selectedHits;

451std::array<float, 6> PMTHitCluster::GetBetaArray()
452{
453 std::array<float, 6> beta = {0., 0., 0., 0., 0., 0.};
454 int nHits = fElement.size();
455
456 if (!fHasVertex) {
457 std::cerr << "PMTHitCluster::GetBetaArray : the hit cluster has no set vertex. Returning a 0-filled array...\n";
458 return beta;
459 }
460
461 if (!nHits) {
462 std::cerr << "PMTHitCluster::GetBetaArray : the hit cluster is empty. Returning a 0-filled array...\n";
463 return beta;
464 }
465
466 for (int i = 0; i < nHits-1; i++) {
467 for (int j = i+1; j < nHits; j++) {
468 // cosine angle between two consecutive uv vectors
469 float cosTheta = fElement.at(i).GetDirection().Dot(fElement.at(j).GetDirection());
470 for (int k = 1; k <= 5; k++)
471 beta[k] += GetLegendreP(k, cosTheta);
472 }
473 }
474
475 for (int k = 1; k <= 5; k++)
476 beta[k] = 2.*beta[k] / float(nHits) / float(nHits-1);
477
478 // Return calculated beta array
479 return beta;
480}

651float PMTHitCluster::GetBurstSignificance(float tBurstWindow)
652{
653 int obs = GetNBurst();

655 float flatDarkRatio = 0.5;
656 for (auto const& hit: fElement) {
