254void PMTHitCluster::ApplyDeadtime(Float deadtime)
255{
256 TVector3 tempVertex;
257 bool bHadVertex = false;
258 if (fHasVertex) {
259 tempVertex = fVertex;
260 RemoveVertex();
261 bHadVertex = true;
262 }
263
264 std::array<Float, MAXPM> hitTime;
265 hitTime.fill(std::numeric_limits<Float>::lowest());
266
267 std::vector<PMTHit> dtCorrectedHits;
268
269 if (!fIsSorted) Sort();
270 for (auto const& hit: fElement) {
271 int hitPMTID = hit.i();
272 if (1 <= hitPMTID && hitPMTID <= MAXPM) {
273 if (hit.t() - hitTime[hitPMTID] > deadtime) {
274 dtCorrectedHits.push_back(hit);
275 hitTime[hitPMTID] = hit.t();
276 }
277 }
278 }
279
280 fElement = dtCorrectedHits;
281
282 if (bHadVertex)
283 SetVertex(tempVertex);

286std::array<float, 6> PMTHitCluster::GetBetaArray()
287{
288 std::array<float, 6> beta = {0., 0., 0., 0., 0., 0.};
289 int nHits = fElement.size();
290
291 if (!fHasVertex) {
292 std::cerr << "PMTHitCluster::GetBetaArray : the hit cluster has no set vertex. Returning a 0-filled array...\n";
293 return beta;
294 }
295
296 if (!nHits) {
297 std::cerr << "PMTHitCluster::GetBetaArray : the hit cluster is empty. Returning a 0-filled array...\n";
298 return beta;
299 }
300
301 for (int i = 0; i < nHits-1; i++) {
302 for (int j = i+1; j < nHits; j++) {
303 // cosine angle between two consecutive uv vectors
304 float cosTheta = fElement[i].GetDirection().Dot(fElement[j].GetDirection());
305 for (int k = 1; k <= 5; k++)
306 beta[k] += GetLegendreP(k, cosTheta);
307 }
308 }
309
310 for (int k = 1; k <= 5; k++)
311 beta[k] = 2.*beta[k] / float(nHits) / float(nHits-1);
312
313 // Return calculated beta array
314 return beta;
315}
