15void ParticleCluster::ReadCommonBlock(vcwork_common primaryCommon, secndprt_common secondaryCommon)
16{
17 Clear();
18
19 // Primaries
20 for (int iVec = 0; iVec < primaryCommon.nvect; iVec++) {
21 int g3pid = primaryCommon.ip[iVec];
22 int g4pid = 0;
23 if (g3pid == 4) {
24 // get pid from neut vectors
25 float posnu[3]; nerdnebk_(posnu);
26 g4pid = nework_.ipne[iVec];
27 }
28 else g4pid = gG3toG4PIDMap[g3pid];
29 Particle primary(g4pid, 0, TVector3(primaryCommon.pos), TVector3(primaryCommon.pin[iVec]));
30 Append(primary);
31 }
32
33 // Secondaries
34 for (int iSec = 0; iSec < secondaryCommon.nscndprt; iSec++) {
35
36 int detsimCode = secondaryCommon.iprtscnd[iSec];
37 int pdgCode = detsimCode;
38 if (detsimCode > 100000)
39 pdgCode = gG3toG4PIDMap[detsimCode];
40 if (detsimCode > 1e9) // nucleus code straight from skg4
41 pdgCode = detsimCode;
42
43 Particle secondary(pdgCode,
44 secondaryCommon.tscnd[iSec]*1e-3,
45 TVector3(secondaryCommon.vtxscnd[iSec]),
46 TVector3(secondaryCommon.pscnd[iSec]),
47 secondaryCommon.iprntprt[iSec],
48 secondaryCommon.lmecscnd[iSec]);
49
50 Append(secondary);
51 }
52
53 Sort();
54}
