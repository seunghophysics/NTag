114void BonsaiManager::FitSKLOWE(const PMTHitCluster& hitCluster)
115{
116 // clear sktq
117 for (int iPMT=0; iPMT<MAXPM; iPMT++) {
118 skt_.tisk[iPMT] = 0;
119 skq_.qisk[iPMT] = 0;
120 skchnl_.ihcab[iPMT] = 0;

122
123 // hitCluster->sktq
124 // multiple hits to the same PMT should be counted as one
125 int iHit = 0;
126 float maxQ = 0;
127 for (auto const& hit: hitCluster) {
128 int iPMT = hit.i()-1;
129 if (!skt_.tisk[iPMT]) {
130 skt_.tisk[iPMT] = hit.t();
131 skchnl_.ihcab[iHit] = hit.i();
132 iHit++;
133 }
134 skq_.qisk[iPMT] += hit.q();
135
136 // mxqisk: ID of PMT with max Q
137 if (skq_.qisk[iPMT] > maxQ)
138 skq_.mxqisk = iPMT;
139 }
140 // nqisk: total number of hit PMTs
141 skq_.nqisk = iHit+1;
142
143 // lfallfit_sk4_data / mc
144 float waterTransparency = 12431.3;
145 int NHITCUT = 1100;
146 int fitFlag; //int flagSkip=0; int flagLog=1;
147 //skheadg_.sk_geometry = 6;
148 //lfallfit_sk6_data_(&waterTransparency, &NHITCUT, &flagSkip, &flagLog, &fitFlag);
149 if (skhead_.nrunsk == 999999)
150 lfallfit_sk4_mc_(&waterTransparency, &NHITCUT, &fitFlag);
151 else
152 lfallfit_sk4_data_(&waterTransparency, &NHITCUT, &fitFlag);
153
154 // retreive common block
155 fFitVertex = TVector3(skroot_lowe_.bsvertex[0], skroot_lowe_.bsvertex[1], skroot_lowe_.bsvertex[2]);
156 fFitTime = skroot_lowe_.bsvertex[3];
157 fFitEnergy = skroot_lowe_.bsenergy;
158 fFitGoodness = skroot_lowe_.bsgood[1];
