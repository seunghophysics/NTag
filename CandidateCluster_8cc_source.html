95void CandidateCluster::FillVectorMap()
96{
97 if (!GetSize()) {
98 //std::cerr << "No elements in CandidateCluster... skipping CandidateCluster::FillVectorMap." << std::endl;
99 }
100
101 else {
102 //auto baseFeatureMap = element[0].GetFeatureMap();
103 bool areFeaturesIdentical = true;
104 for (unsigned int iCandidate = 0; iCandidate < GetSize(); iCandidate++) {
105 auto comparedFeatureMap = fElement[iCandidate].GetFeatureMap();
106
107 for (auto const& basePair: fFeatureVectorMap) {
108
109 if (!comparedFeatureMap.count(basePair.first)) {
110 std::cerr << "Registered key " << basePair.first << " not found in candidate!" << std::endl;
111 areFeaturesIdentical = false;
112 }
113 else {
114 fFeatureVectorMap[basePair.first]->resize(iCandidate);
115 fFeatureVectorMap[basePair.first]->push_back(comparedFeatureMap[basePair.first]);
116 }
117 }
118
119 for (auto const& comparedPair: comparedFeatureMap) {
120 if (!fFeatureVectorMap.count(comparedPair.first)) {
121 std::cerr << "Candidate key " << comparedPair.first << " not found in registered keys!" << std::endl;
122 areFeaturesIdentical = false;
123 }
124 }
125 }
126
127 if (!areFeaturesIdentical) {
128 std::cerr << "Make sure all candidates share the same set of features"
129 " specified by CandidateCluster::RegisterFeatureNames!" << std::endl;
130 }
131
132 assert(fFeatureVectorMap.begin()->second->size() == GetSize());
133 }
