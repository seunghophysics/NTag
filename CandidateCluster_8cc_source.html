29void CandidateCluster::DumpAllElements(std::vector<std::string> keys, bool showTaggedOnly) const
30{
31 Printer msg;
32 msg.PrintBlock(fName + (showTaggedOnly?" Tagged":"") + " Candidates", pSUBEVENT);
33
34 if (!GetSize()) {
35 msg.Print("No candidate in cluster!");
36 }
37 else {
38 std::cout << "\033[4m No. ";
39 auto baseFeatureMap = fElement.at(0).GetFeatureMap();
40
41 if (keys.empty()) {
42 for (auto const& pair: baseFeatureMap)
43 keys.push_back(pair.first);
44 }
45
46 for (auto const& key: keys) {
47 int textWidth = key.size()>6 ? key.size() : 6;
48 std::cout << std::right << std::setw(textWidth) << key << " ";
49 }
50 std::cout << "\033[0m\n";
51
52 for (unsigned int iCandidate = 0; iCandidate < GetSize(); iCandidate++) {
53
54 if (showTaggedOnly && fElement.at(iCandidate).Get("TagClass")==0) continue;
55
56 std::cout << std::right << std::setw(4) << iCandidate+1 << " ";
57 auto candidateFeatureMap = fElement.at(iCandidate).GetFeatureMap();
58 for (auto const& key: keys) {
59 int textWidth = key.size()>6 ? key.size() : 6;
60 float value = fElement.at(iCandidate)[key];
61 auto keyString = TString(key);
62 if (keyString.Contains("Index")) {
63 std::cout << std::right << std::setw(textWidth) << (value>=0 ? std::to_string(int(value+1)) : "-") << " ";
64 }
65 else if (keyString.Contains("FitT") && fabs(value) < 10) {
66 std::cout << std::fixed << std::setprecision(2) << std::setw(textWidth) << value << " ";
67 }
68 else if (keyString.Contains("BSenergy")) {
69 std::cout << std::setw(textWidth) << (value>=0 ? Form("%3.2f",value) : "-");
70 }
71 else if (key == "Label") {
72 if (value <= lNoise) std::cout << std::right << std::setw(textWidth) << "-";
73 else if (value <= lDecayE) std::cout << std::right << std::setw(textWidth) << "e";
74 else if (value <= lnH) std::cout << std::right << std::setw(textWidth) << "nH";
75 else if (value <= lnGd) std::cout << std::right << std::setw(textWidth) << "nGd";
76 else if (value <= lGamma) std::cout << std::right << std::setw(textWidth) << "g";
77 else if (value <= lRemnant) std::cout << std::right << std::setw(textWidth) << "=";
78 else if (value <= lUndefined) std::cout << std::right << std::setw(textWidth) << "?";
79 std::cout << " ";
80 }
81 else if (key == "TagClass") {
82 if (value == typeE) std::cout << std::right << std::setw(textWidth) << "e";
83 else if (value == typeN) std::cout << std::right << std::setw(textWidth) << "n";
84 else std::cout << std::right << std::setw(textWidth) << "-";
85 }
86 else if (fabs(value) < 1 && value != 0) {
87 value = roundf(value*100)/100;

89 }

102void CandidateCluster::FillVectorMap()
103{
104 if (!GetSize()) {
105 //std::cerr << "No elements in CandidateCluster... skipping CandidateCluster::FillVectorMap." << std::endl;
106 }
107
108 else {
109 //auto baseFeatureMap = element[0].GetFeatureMap();
110 bool areFeaturesIdentical = true;
111 for (unsigned int iCandidate = 0; iCandidate < GetSize(); iCandidate++) {
112 auto comparedFeatureMap = fElement.at(iCandidate).GetFeatureMap();
113
114 for (auto const& basePair: fFeatureVectorMap) {
115
116 if (!comparedFeatureMap.count(basePair.first)) {
117 std::cerr << "Registered key " << basePair.first << " not found in candidate!" << std::endl;
118 areFeaturesIdentical = false;
119 }
120 else {
121 fFeatureVectorMap[basePair.first]->resize(iCandidate);
122 auto value = comparedFeatureMap[basePair.first];
123 if (std::isnan(value) || std::isinf(value)) {
124 if (std::isnan(value)) {
125 std::cerr << Form("Candidate #%d: key %s value is NaN!", iCandidate, basePair.first.c_str()) << std::endl;
126 }
127 if (std::isinf(value)) {
128 std::cerr << Form("Candidate #%d: key %s value is inf!", iCandidate, basePair.first.c_str()) << std::endl;
129 }
130 std::cerr << Form("Dumping all features in map...", basePair.first) << std::endl;
131 DumpAllElements(gNTagFeatures);
132 abort();
133 }
134 else
135 fFeatureVectorMap[basePair.first]->push_back(value);
136 }
137 }
138
139 for (auto const& comparedPair: comparedFeatureMap) {
140 if (!fFeatureVectorMap.count(comparedPair.first)) {
141 std::cerr << "Candidate key " << comparedPair.first << " not found in registered keys!" << std::endl;
142 areFeaturesIdentical = false;
143 }
144 }
145 }
146
147 if (!areFeaturesIdentical) {
148 std::cerr << "Make sure all candidates share the same set of features"
149 " specified by CandidateCluster::RegisterFeatureNames!" << std::endl;
150 }
151
152 assert(fFeatureVectorMap.begin()->second->size() == GetSize());
153 }
154
155 fNCandidates = GetSize();
156}
