44float GetRMS(const std::vector<float>& vec)
45{
46 float N = static_cast<float>(vec.size());
47 float mean = 0.;
48 float var = 0.;
49
50 for (auto const& value: vec)
51 mean += value / N;
52 for (auto const& value: vec)
53 var += (value-mean)*(value-mean) / (N-1);
54
55 return sqrt(var);
56}

78float GetOpeningAngle(TVector3 uA, TVector3 uB, TVector3 uC)
79{
80 // make sure the inputs are unit vectors
81 // uA = uA.Unit(); uB = uB.Unit(); uC = uC.Unit();
82
83 // sides of the triangle formed by the three unit vectors
84 double a = (uA-uB).Mag();
85 double b = (uC-uA).Mag();
86 double c = (uB-uC).Mag();
87
88 if (a*b*c == 0) {
89 //double angleAB = (180./M_PI) * uA.Angle(uB)/2.;
90 //double angleAC = (180./M_PI) * uA.Angle(uC)/2.;
91 return uA.Angle(uB) == 0 ? (uA.Angle(uC) == 0 ? 0 : uA.Angle(uC)) : uA.Angle(uB);
92 }
93
94 else {
95 // circumradius of the triangle
96 double r = a*b*c / sqrt((a+b+c)*(-a+b+c)*(a-b+c)*(a+b-c));
97
98 if (r>=1)
99 return 90.; // prevents NaN
100 else
101 return (180./M_PI) * asin(r);
102 }
103}

111float GetDWallInDirection(TVector3 vtx, TVector3 dir)
112{
113 dir = dir.Unit();
114
115 float dot = vtx.Dot(dir) - vtx.z()*dir.z();
116 float dirSq = dir.Perp2(); float vtxSq = vtx.Perp2();
117
118 // Calculate distance to barrel and distance to top/bottom
119 float distR = (-dot + sqrt(dot*dot + dirSq*(RINTK*RINTK - vtxSq))) / dirSq;
120 float distZ = dir.z() > 0 ? (ZPINTK-vtx.z())/dir.z() : (ZMINTK-vtx.z())/dir.z();
121
122 // Return the smaller
123 return distR < distZ ? distR : distZ;
124}

225std::vector<std::pair<float, int>> Histogram(std::vector<float> vec, int nBins, float min, float max)
226{
227 std::vector<std::pair<float, int>> hist(nBins, std::pair<float, int>(0, 0));
228 const float bWidth = (max - min) / float(nBins);
229
230 std::sort(vec.begin(), vec.end());
231
232 int index = 0;
233 auto binStartItr = vec.begin();
234 for (auto& bin: hist) {
235 const float bMin = min + index*bWidth;
236 const float bMax = min + (index+1)*bWidth;
237
238 auto start = std::lower_bound(binStartItr, vec.end(), bMin);
239 auto end = std::lower_bound(start, vec.end(), bMax*0.99999);
240 binStartItr = end;
241
242 bin.first = (bMin+bMax)/2.; // bin center
243 bin.second = end - start; // bin value
244 index++;
245 }
246
247 return hist;
248}
