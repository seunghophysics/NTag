51float GetRMS(const std::vector<float>& vec)
52{
53 float N = static_cast<float>(vec.size());
54 float mean = 0.;
55 float var = 0.;
56
57 for (auto const& value: vec)
58 mean += value / N;
59 for (auto const& value: vec)
60 var += (value-mean)*(value-mean) / (N-1);
61
62 return sqrt(var);
63}

85float GetOpeningAngle(TVector3 uA, TVector3 uB, TVector3 uC)
86{
87 // make sure the inputs are unit vectors
88 uA = uA.Unit(); uB = uB.Unit(); uC = uC.Unit();
89
90 //float ax = uA.x(); float ay = uA.y(); float az = uA.z();
91 //float bx = uB.x(); float by = uB.y(); float bz = uB.z();
92 //float cx = uC.x(); float cy = uC.y(); float cz = uC.z();
93
94 // sides of the triangle formed by the three unit vectors
95 double a = (uA-uB).Mag();
96 double b = (uC-uA).Mag();
97 double c = (uB-uC).Mag();
98
99 if (fabs(a*b*c) < 1e-15) {
100 double angleAB = uA.Angle(uB);
101 double angleAC = uA.Angle(uC);
102 return (180./M_PI) * (angleAB ? (angleAC ? 0 : angleAC) : angleAB);
103 }
104
105 else {
106 // circumradius of the triangle
107 double r = a*b*c / sqrt((a+b+c)*(-a+b+c)*(a-b+c)*(a+b-c));
108
109 if (r>=1)
110 return 90.; // prevents NaN
111 else
112 return (180./M_PI) * asin(r);
113 }
114}

122float GetDWallInDirection(TVector3 vtx, TVector3 dir)
123{
124 assert(dir.Mag()>0);
125 dir = dir.Unit();
126
127 float dot = vtx.Dot(dir) - vtx.z()*dir.z();
128 float dirSq = dir.Perp2(); float vtxSq = vtx.Perp2();
129 //float dirx = dir.x(); float diry = dir.y(); float dirz = dir.z();
130
131 // Calculate distance to barrel and distance to top/bottom
132 float distR = fabs((-dot + sqrt(dot*dot + dirSq*(RINTK*RINTK - vtxSq))) / dirSq);
133 float distZ = fabs(dir.z() > 0 ? (ZPINTK-vtx.z())/dir.z() : (ZMINTK-vtx.z())/dir.z());
134
135 // Return the smaller
136 if (std::isnan(distZ)) return distR;
137 else return distR < distZ ? distR : distZ;
138}

223std::vector<std::pair<float, int>> Histogram(std::vector<float> vec, int nBins, float min, float max)
224{
225 std::vector<std::pair<float, int>> hist(nBins, std::pair<float, int>(0, 0));
226 const float bWidth = (max - min) / float(nBins);
227
228 std::sort(vec.begin(), vec.end());
229
230 int index = 0;
231 auto binStartItr = vec.begin();
232 for (auto& bin: hist) {
233 const float bMin = min + index*bWidth;
234 const float bMax = min + (index+1)*bWidth;
235
236 auto start = std::lower_bound(binStartItr, vec.end(), bMin);
237 auto end = std::lower_bound(start, vec.end(), bMax*0.99999);
238 binStartItr = end;
239
240 bin.first = (bMin+bMax)/2.; // bin center
241 bin.second = end - start; // bin value
242 index++;
243 }
244
245 return hist;
246}
