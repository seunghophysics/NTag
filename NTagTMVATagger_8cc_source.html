46int NTagTMVATagger::Classify(const Candidate& candidate)
47{
48 int tagClass = 0;
49 bool isEarly = candidate.Get("N50", -1) < 0; // muechk flag
50 int nHits = candidate.Get("NHits");
51 float fitT = candidate.Get("FitT");
52 float tmvaOut = isEarly ? /* muechk */ 1 :
53 /* ntag */ GetLikelihood(candidate);
54
55 // simple cuts mode for e/n separation
56 if (fDoTagE) {
57 if (tmvaOut < TAGOUTCUT) tagClass = typeMissed;
58 else if (nHits > E_NHITSCUT && fitT < E_TIMECUT) tagClass = typeE;
59 else tagClass = typeN;
60 }
61 // naive tagging mode without e/n separation
62 else {
63 if (isEarly) tagClass = typeE; // e: muechk
64 else if (tmvaOut > TAGOUTCUT) tagClass = typeN; // n: ntag && out cut
65 else tagClass = typeMissed; // otherwise noise
66 }
67
68 return tagClass;
69}
