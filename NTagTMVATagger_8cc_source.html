48int NTagTMVATagger::Classify(const Candidate& candidate)
49{
50 int tagClass = typeMissed;
51 bool isEarly = candidate.Get("N50", -1) < 0; // muechk flag
52 int nHits = candidate.Get("NHits");
53 float fitT = candidate.Get("FitT");
54 float tmvaOut = isEarly ? /* muechk */ 1 :
55 /* ntag */ GetLikelihood(candidate);
56
57 // simple cuts mode for e/n separation
58 if (fDoTagE) {
59 if (tmvaOut < TAGOUTCUT) tagClass = typeMissed;
60 else if (nHits > E_NHITSCUT && fitT < E_TIMECUT) tagClass = typeE;
61 else tagClass = typeN;
62 }
63 // naive tagging mode without e/n separation
64 else {
65 if (isEarly) tagClass = typeE; // e: muechk
66 else if (tmvaOut > TAGOUTCUT) tagClass = typeN; // n: ntag && out cut
67 else tagClass = typeMissed; // otherwise noise
68 }
69
70 return tagClass;
71}
