14void TaggableCluster::ReadParticleCluster(const ParticleCluster& particleCluster)
15{
16 Clear();
17
18 for (auto const& particle : particleCluster) {
19
20 // skip particles outside tank
21 //if (GetDWall(particle.Vertex()) < 0) continue;
22
23 // gamma-ray produced by n-capture
24 if (particle.IntID() == iNCAPTURE && particle.PID() == GAMMA) {
25
26 bool isNewCapture = true;
27
28 // check saved captures
29 for (unsigned int iCapture = 0; iCapture < fElement.size(); iCapture++) {
30
31 Taggable& capture = fElement[iCapture];
32
33 // if a matching capture time exists,
34 // just add the gamma-ray to the matching capture
35 if (fabs((double)(particle.Time()-capture.Time()))<1.e-4) {
36 isNewCapture = false;
37 capture.SetEnergy(capture.Energy()+particle.Energy());
38 }
39 }
40
41 // if this n-capture gamma-ray is new
42 if (isNewCapture) {
43 Taggable capture(typeN, particle.Time(), particle.Energy(), particle.Vertex());
44 Append(capture);
45 }
46 }
47
48 // decay electrons
49 else if (particle.IntID() == iDECAY && abs(particle.PID()) == ELECTRON)
50 Append(Taggable(typeE, particle.Time(), particle.Energy(), particle.Vertex()));
51 }
52}
